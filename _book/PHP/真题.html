第二节：真题

引用变量

$data = ['a', 'b', 'c'];
foreach($data as $key => $val){
    $val = &$data[$key];
    print_r($data);
}
//['a', 'b', 'c']
//['b', 'b', 'c']
//['b', 'c', 'c']

运算符优先级

<?php
$a = $b = 0;
if($a = 3 > 0 || $b = 3 > 0){
    var_dump($a, $b);
    $a++;
    $b++;
    echo $a . PHP_EOL;
    echo $b . PHP_EOL;
}
//true 0    1 1
//因为||前边true，所以$b=3>0不会走到；true++不影响布尔值

从函数返回一个引用，必须在函数声明和指派返回值给一个变量时都使用引用变量符&

function &func(){
    static $num = 1;
    return $num;
}
$a = func();
echo func();//1
$a = &func();
$a = 10;
echo func();//10
